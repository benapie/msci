\begin{questions}
	\question 
	\begin{parts}
		\part Determine $\Z[i]^\times$.
		\part Factorise $8 + 9i$ into irreducibles in $\Z[i]$.
	\end{parts}

	\question
	Let $K = \Q(\sqrt{-6})$.
	\begin{parts}
		\part
		Show that the element $1 - 3 \sqrt{-6} \in \mathcal O_K$ is irreducible.

		\part Show that the element $16 + 7 \sqrt{-6} \in \mathcal O_K$ is not
		irreducible, and find a factorisation into irreducibles in 
		$\mathcal O_K$.
	\end{parts}

	\question
	Let $K = \Q(\sqrt{-6})$, and let $I$ be the ideal in $\mathcal O_K$
	generated by $2$ and $\sqrt{-6}$, that is 
	$I = (2, \sqrt{-6})_{\mathcal O_K}$.
	Show that $I$ is not a principal ideal.

	\question
	Let $K = \Q(\sqrt{-2})$.
	Show that the ring $\mathcal O_K$ is a Euclidean Domain.

	\question
	Let $R = \Z[\sqrt{-26}]$, and consider the ideal $J$ in $R$ generated by
	$1 + \sqrt{-26}$ and $3$, that is $J = (3, 1 + \sqrt{-26})_R$.
	Show that
	\begin{parts}
		\part 
		$J^3$ is a principal ideal but $J$ is not.

		\part 
		Show further that $J^2$ is not a principal ideal.

		\part
		Show that $J$ is a maximal ideal in $R$.
	\end{parts}

	\question
	Find two units $u$ and $v$ in $\Z[\sqrt 5] \subset R$ such that $u,v > 1$.

	\question
	\begin{parts}
		\part
		Let $R$ be an integral domain that is not a field.
		Show that: if $R$ is a Euclidean domain then there exists a
		$0 \neq u \in R \setminus R^\times$ with the following property:
		for every $x \in R$ there is some $z \in R^\times \cup \left\{
			0
		\right\}$ such that $u \mid x - z$.

		\part
		(Hard) Let $K = \Q(\sqrt{-19})$.
		Show that $\mathcal O_K$ is not a Euclidean Domain.
	\end{parts}
\end{questions}
