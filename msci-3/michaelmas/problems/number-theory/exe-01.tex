\begin{questions}
	\question 
	Let $F$ be a field.
	We define the characteristic of $F$ to be the smallest
	positive integer $n$ such that $n \cdot 1 = 0$,
	and if such a positive does not exist we define the characteristic to be
	zero.
	\begin{parts}
		\part 
		Show that the characteristic of a field is either $0$ or a prime number.

		\part
		Give an example of an infinite field of non-zero characteristic.

		\part Let $F$ be a field of characteristic zero.
		Show that $\Q$, the field of rational numbers, is a subfield of $F$.
	\end{parts}

	\question
	\begin{parts}
		\part
		Show that a polynomial $f(x) \in \Q[x]$ of degree two or three is
		irreducible in $\Q[x]$ if and only if $f(x)$ does not have any root in
		$\Q$.
		Give an example  where the statement fails when $f(x)$ is of degree
		larger than three.

		\part
		Let $p(x) = a_n x^n + a_{n-1} x^{n-1} + \ldots + a_0$ be a polynomial
		of degree $n$ with $a_i \in \Z$ for $i \in \left\{
			0, \ldots, n
		\right\}$. 
		Assume $p\left( 
			\frac rs 
		\right) = 0$ for some $r, s \in \Z$ with $\gcd(r,s) = 1$.
		Show that $r$ divides $a_0$ and $s$ divides $a_n$.

		\part
		Show that the polynomial $f(x) = x^3 - 3x - 1$ is irreducible in
		$\Q[x]$, but it is reducible in $\R[x]$.
	\end{parts}

	\question
	\begin{parts}
		\part
		Determine the units in the rings $\Z[x]$ and $\Q[x]$.

		\part
		Let $f(x)$ be a polynomial with coefficients in $\Z$.
		We recall that Gauss' Lemma says: if $f(x)$ is irreducible in $\Z[x]$,
		then it is also irreducible in $\Q[x]$.
		Show that the converse is not true in general by giving a
		counterexample.

		\part
		Assume not that $f(x) = \sum_{i=0}^n a_i x^i$ with $a_i \in \Z$ and
		$\gcd(a_0, a_1, \ldots, a_n) = 1$.
		Show that in this case the converse of Gauss' Lemma holds.

		\part
		Show that the polynomail $f(x) = x^3 + x + 1$ is irreducible both in
		$\Q[x]$ and $\Z[x]$.
	\end{parts}

	\question
	Show that the complex number $\sqrt 3 + i$ is algebraic over $\Q$.

	\question
	Let $F \subset L$ be a field extension, and let $\alpha \in L$ be algebraic
	over $F$.
	Write $p(x) \in F[x]$ for the minimal polynomial of $\alpha$ over $F$.
	Show that
	\begin{parts}
		\part if we write $I = (p(x))$ for the ideal of $F[x]$ generated by
		$p(x)$ then the quotient ring $F[x]/I$ is a field; and

		\part the fields $F(\alpha)$ and $F[x]/I$ are isomorphic.
	\end{parts}
\end{questions}
