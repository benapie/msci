        No. Consider the following graph.
        \begin{center}
            \begin{tikzpicture}
                \node[main] (1) {};
                \node[main] (2) [above of=1] {};
                \node[main] (3) [right of=2] {};
                \node[main] (4) [right of=1] {};
                \node[main] (5) [above of=2] {};
                \draw (1) -- (2) (3) -- (4) -- (1) -- (3);
                \draw (2) -- (4);
                \draw (2) -- (5);
                \draw (5) -- (3);
            \end{tikzpicture}
        \end{center}
    \end{solution}
