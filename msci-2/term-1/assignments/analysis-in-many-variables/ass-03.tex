\setcounter{question}{11}
\question Compute the divergence, $\nabla \cdot \bm A$, of the following vector fields:
\begin{parts}
    \part $\bm A(x, y, z) = yz \bm e_1 + xz \bm e_2 + xy \bm e_3$;
    \begin{solution}
        \begin{align*}
            \nabla \cdot \bm A &= \frac{\partial}{\partial x} (yz) + \frac{\partial}{\partial y} (xz) + \frac{\partial}{\partial z} (xy) \\
                               &= 0.
        \end{align*}
    \end{solution}

    \part $\bm A(x, y, z) = (x^2 + y^2 + z^2)(3\bm e_1 + 4\bm e_2 + 5\bm e_3)$; and
    \begin{solution}
        \begin{align*}
            \nabla \cdot \bm A &= 3 (2x) + 4 (2y) + 5 (2z) \\
                               &= 6x + 8y + 10z.
        \end{align*}
    \end{solution}

    \part $\bm A(x, y, z) = (x + y)\bm e_1 + (y + z)\bm e_2 + (z + x)\bm e_3$.
    \begin{solution}
        \begin{align*}
            \nabla \cdot \bm A &= 1 + 1 + 1 \\
                               &= 3.
        \end{align*}
    \end{solution}
\end{parts}

\setcounter{question}{13}
\question If $f(r)$ is a differentiable function of $r = \norm{\bm x}$, show that
\begin{parts}
    \part $\grad{f(r)} = \dfrac{f'(r) \bm x}{r}$; and
    \begin{solution}
        \begin{align*}
            \nabla f(r) &= \bm e_1 \frac{\partial f}{\partial x} + \bm e_2 \frac{\partial f}{\partial y} + \bm e_3 \frac{\partial f}{\partial z} \\
                        &= \bm e_1 \frac{df}{dr} \frac{\partial r}{\partial x} + \bm e_2 \frac{df}{dr} \frac{\partial r}{\partial y} + \bm e_3 \frac{df}{dr} \frac{\partial r}{\partial z} \\
            \frac{\partial r}{\partial x_i} &= \frac{x_i}{r} \\
            \nabla f    &= \frac{f'(r) \bm x}{r} 
        \end{align*}
    \end{solution}

    \part $\curl{(f(r) \bm x)} = 0$.
    \begin{solution}
        \begin{align*}
            \nabla \times (f \bm x) &= (\nabla f) \times \bm x + f(\nabla \times \bm x) \\
            \nabla \times \bm x     &= \bm e_1 \left(\frac{\partial z}{\partial y} - \frac{\partial y}{\partial z}\right) - \bm e_2 \left(\frac{\partial z}{\partial x} - \frac{\partial x}{\partial z}\right) + \bm e_3 \left(\frac{\partial y}{\partial x} - \frac{\partial x}{\partial y}\right) = \bm 0 \\
            \nabla \times (f \bm x) &= \frac{f'(r) \bm x}{r} \times \bm x + \bm 0 \\
                                    &= \bm 0
        \end{align*}
        as $\bm x \times \bm x = \bm 0$.
    \end{solution}
\end{parts}
