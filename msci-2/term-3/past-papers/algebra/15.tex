\question
Let $H$ be a subgroup of $S_4$ containing at least $10$ elements.
\begin{parts}
    \part
    Assuming $H$ is a proper subgroup,
    what is the order of $H$?
    Justify.

    \part 
    How many transpositions (i.e. $2$-cycles)
    does $S_4$ have?

    \part 
    Show that if $H$ is a proper subgroup, 
    then it is not possible for $H$ to contain every
    transposition of $S_4$.
\end{parts}

\question
Consider the equation
$y^2 = y + 1$.
\begin{parts}
    \part
    Show that the above equation has no solution for
    $y$ in the ring $\Z/3$.

    \part
    Consider the ideal 
    $I = \langle x^2 - x - 1 \rangle$ of $\Z/3[x]$.
    Show that the element $x + I$ of the quotient ring
    $\Z/3[x]/I$ is a solution to the above equation.

    \part
    With the ideal $I$ from the previous part,
    show that $1$ is not contained in $I$ and deduce
    that the quotient ring $\Z/3[x]/I$ is not the zero ring.
\end{parts}

\question
Consider the mapping
$f: \Z \to S_4$, $f(n) = (1\;2\;3\;4)^n$.
\begin{parts}
    \part
    Show that $f$ is a group homomorphism.

    \part
    Determine the kernel of $f$.
    Justify rigorously.

    \part
    The First Isomorphism Theorem guarantees the existence of a
    certain isomorphism $\ov f$ related to $f$.
    What is the domain of $\ov f$, and which element of the
    domain is mapped to $(1\;2\;3\;4) \in S_4$ under $\ov f$?
\end{parts}

\question
\begin{parts}
    \part
    Let $C$ be a cyclic group of even order $n$ 
    generated by $g$. 
    Show that the element $g^{n/2}$ 
    has order $2$ in $C$.
    
    \part
    Assume $D$ is an integral domain. 
    Show that the element $1$ has 
    at most two square roots 
    (elements whose square is $1$) in $D$.

    \part
    Assume $F$ is a finite field of size $q$. 
    What is the order of the multiplicative 
    group of units of $F$? Justify.

    \part
    Assume $F$ is a finite field of odd size $q$. 
    It is known that the group of units 
    of any finite field is cyclic. 
    Based on the previous parts, 
    show that the number of square roots 
    of $1$ in $F$ is exactly two.

    \part
    Show that if a finite field $F$ has even size, 
    then there is only one square root of $1$ in $F$.
\end{parts}

\question
Let $K$ be a field, and let $P$ be a polynomial in $K[x]$.
\begin{parts}
    \part
    Let $Q \in K[x]$. 
    Show that $P$ is divisible by $Q$ 
    if and only if $\langle P \rangle \subset \langle Q \rangle$. 

    \part
    Show that if $Q \in K[x]$ 
    is a non-constant polynomial, 
    then $\langle Q \rangle \neq K[x]$.

    \part
    Using the previous parts, 
    show that if $\langle P \rangle$ 
    is a maximal ideal of $K[x]$, 
    then $P$ is irreducible.

    \part
    Based on part (c), 
    if the quotient ring $K[x]/\langle P \rangle$ 
    is a field, 
    what can you deduce about the polynomial $P$?
\end{parts}

\question
Assume $G$ is a group, 
$H$ a subgroup of $G$ and $N$ 
a normal subgroup of $G$. 
Consider the set 
$HN = \{hn : h \in H,n \in N\}$.
\begin{parts}
    \part
    Show that the set $H \cap N$ is a normal subgroup of $H$.

    \part
    Let $h \in H$ and $n \in N$. 
    Show that $nh=hn'$ for some $n' \in N$.

    \part
    With the help of the previous part, 
    prove that $HN$ is a subgroup of $G$.

    \part
    Assume known that $N$ is normal 
    in $HN$ and $f(h) = hN$ is a surjective 
    homomorphism from $H$ to $HN/N$. 
    Show that the groups $H/(H \cap N)$ 
    and $HN/N$ are isomorphic.
\end{parts}

\question
Consider an action of a group $G$ on a set $X$ with a subset $Y$.
\begin{parts}
    \part
    Show that the stabiliser $\Stab(Y)$ is closed under multiplication.

    \part
    Assume $Y$ is finite, 
    and let $y \in Y$ and $g \in \Stab(Y)$ . 
    Prove that $g^ky = y$ 
    for some positive $k \in \Z$. 
    Deduce that $g^{-1}y = g^my$ 
    for some non-negative $m \in \Z$.

    \part
    Using the previous parts, 
    show that if $Y$ is finite, then $\Stab(Y)$ is a subgroup of $G$.

    \part
    Consider the case $G=X=\Z$,
    with action $gx=g+x$. 
    Find a subset $Y$ of $X$, 
    such that $GY$ is not a subgroup of $G$. 
    Justify your findings.
\end{parts}
