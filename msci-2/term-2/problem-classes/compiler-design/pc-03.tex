\question Consider the following simple grammar:
\begin{align*}
    T &\to R            \\
    T &\to aTc          \\
    R &\to \varepsilon \\
    R &\to RbR.
\end{align*}
\begin{parts}
    \part Construct a leftmost and rightmost derivation of the string $aabbbcc$.
    \begin{solution}
        Leftmost:
        \begin{align*}
            T 
            &\implies aTc \\
            &\implies aaTcc \\
            &\implies aaRcc \\
            &\implies aaRbRcc \\
            &\implies aaRbRbRcc \\
            &\implies aaRbRbRbRcc \\
            &\implies aabRbRbRcc \\
            &\implies aabbRbRcc \\
            &\implies aabbbRcc \\
            &\implies aabbbcc.
        \end{align*}
        Rightmost:
        \begin{align*}
            T 
            &\implies aTc \\
            &\implies aaTcc \\
            &\implies aaRcc \\
            &\implies aaRbRcc \\
            &\implies aaRbRbRcc \\
            &\implies aaRbRbRbRcc \\
            &\implies aaRbRbRbcc \\
            &\implies aaRbRbbcc \\
            &\implies aaRbbbcc \\
            &\implies aabbbcc.
        \end{align*}
    \end{solution}

\question Consider the grammar
\[
    A \to Aab \mid Aaab \mid cA \mid bcab.
\]
\begin{parts}
    \part Eliminate the left recursion in the grammar.
    \begin{solution}
        \begin{align*}
            A &\to bcabB \mid cAB \\
            B &\to abB \mid aabB \mid \varepsilon.
        \end{align*}
    \end{solution}

    \part After the elimination of the left recursion in this grammar,
    perform left factoring in the resulting grammar.
    \begin{solution}
        \begin{align*}
            A &\to bcabB \mid cAB \\
            B &\to aC \mid \varepsilon \\
            C &\to bB \mid abB.
        \end{align*}
    \end{solution}
\end{parts}
