\setcounter{question}{97}
\question State a necessary and sufficient condition for a vector field $\bm F$
to be expressible in the form $\bm F = \nabla \phi$ in some simply-connected region.
The scalar field $\phi$ is called a \emph{scalar potential}, though sometimes the
opposite sign is used.

\begin{parts}
    \part Determine whether the following vector fields are expressible in this form
    \begin{subparts}
        \subpart $(\bm a \cdot \bm x) \bm x$;

        \subpart $(\bm a \cdot \bm x) \bm a$;

        \subpart $(\bm a \cdot \bm a) \bm x$;

        \subpart $\bm a \times \bm x$; and

        \subpart $\bm a \times (\bm a \times \bm x)$.
    \end{subparts}
    
    \part Find the vector fields for which the corresponding potentials are
    \begin{subparts}
        \subpart $\frac12(\bm a \cdot \bm x)^2$; and
        
        \subpart $\frac12 a^2 \norm{\bm x}^2 - \frac12(\bm a \cdot \bm x)^2$.
    \end{subparts}
\end{parts}

\setcounter{question}{99}
\question \hfill
\begin{parts}
    \part State the divergence theorem.

    \part Use it to calculate
    \[
        \int_S \bm U \cdot d\bm A
    \]
    where $S$ is the surface of the cube with faces
    $x = \pm 1$,
    $y = \pm 1$, and
    $z = \pm 1$ and
    \[
        \bm U(x,y,z) = (xy^2, yx^2, z).
    \]

    \part Check your answer by a direct calculation of the surface integral.
\end{parts}
